{
        List<Integer> res = new ArrayList<>();
        Queue<TreeNode> queue = new LinkedList<>();
        // 注意判断控，如果为null，下面的queue不会为空
        if (root == null) {
            return res;
        }
        queue.offer(root);
        // 判断条件不能是queue!=null
        while (!queue.isEmpty()) {
            List<Integer> list = new ArrayList<>();
            // 每次只遍历当前层册
            int k = queue.size();
            for (int i = 0; i < k; i++) {
                TreeNode poll = queue.poll();
                if (i== k-1) {
                    res.add(poll.val);
                }
                if (poll.left != null) {
                    queue.offer(poll.left);
                }
                if (poll.right!= null) {
                    queue.offer(poll.right);
                }
            }
        }
        return res;
    }
