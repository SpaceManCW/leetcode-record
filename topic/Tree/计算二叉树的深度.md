class Solution {
    public int calculateDepth(TreeNode root) {
        return calculateDepth1(root);
    }
    public int calculateDepth1(TreeNode root) {
        if (root == null) return 0;
        return Math.max(calculateDepth1(root.left) + 1 , calculateDepth1(root.right) + 1);
    }

}
