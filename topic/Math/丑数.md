// 235一直除就可以
class Solution {
    public boolean isUgly(int n) {
        if (n <= 0 ) return false;
        int[] ns = new int[]{2,3,5};
        for (int i : ns) {
            while ( n % i == 0) {
                n/=i;
            }
        }
        return n==1;
    }
}
