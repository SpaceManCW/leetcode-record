class Solution {
    public int reverse(int x) {
        long res = 0;
        int fu = 1;
        if (x < 0) {
            x = x * (-1);
            fu = -1;
        }
        if (x == 0) return 0;
        while (x > 0) {
            res = res * 10 + x % 10;
            x /= 10;
        }
        if (res > 2147483647) {
            return 0;
        }
        return (int) res * fu;
    }
}
