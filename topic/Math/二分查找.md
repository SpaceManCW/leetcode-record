class Solution {
    public int search(int[] nums, int target) {
            int i= 0;
            // j的赋值必须要-1 要不然会出现数组越界
            int j = nums.length - 1;
            while (i <= j) {
                int m = i + (j-i)/2;
                if (nums[m]> target) {
                    // 这个条件是 i>j 结束循环的关键，因为这个数组里面不一定包含targer，所以必须要指针向后移动
                    // 例如 [1,3] 找2的话就会出现问题
                    // 如果一定会存在的话就不需要指针移动
                    j = m - 1;
                }
                if (nums[m] < target) {
                    i = m + 1;
                }
                if (nums[m] == target) {
                    return m;
                }
            }
            return -1;
    }
}
