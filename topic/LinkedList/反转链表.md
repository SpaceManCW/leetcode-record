    class Solution {
        public ListNode reverseList(ListNode head) {
            ListNode h = new ListNode(0);
            h.next = head;
            while (head != null && head.next != null) {
                ListNode p = head.next;
                head.next = head.next.next;
                p.next = h.next;
                h.next = p;
            }
            return h.next;
        }
    }
