### 2024.5.13 环形链表  https://leetcode.cn/problems/linked-list-cycle/description/

---

### JS

```js
// 把遍历过的节点打上标记  再次遇到就是有环
var hasCycle = function(head) {
    while(head) {
        if(head.tag) {
            return true
        }
        head.tag = true
        head = head.next
    }
    return false
};
```
