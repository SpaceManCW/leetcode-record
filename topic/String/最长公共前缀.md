class Solution {
    public String longestCommonPrefix(String[] strs) {
        if (strs.length <= 0) return "";
        int k = strs[0].length();
        for (int i = 0; i < k; i++) {
            char c = strs[0].charAt(i);
            for (int j = 0; j < strs.length; j++) {
                // 这层循环前面的判断必须带等号
                if (strs[j].length() <= i || strs[j].charAt(i) != c) {
                    return strs[0].substring(0,i);
                }
            }
        }
        return strs[0];
    }
}
