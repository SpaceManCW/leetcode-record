### 2024.5.20 最后一个单词的长度  https://leetcode.cn/problems/length-of-last-word/

---

### **JS**

```js
var lengthOfLastWord = function(s) {
    const arr = Array.from(s)
    let res = 0, islast = true
    for(let i = arr.length - 1; i >= 0; i--) {
        if(arr[i] !== ' ') {
            islast = false
            res++
        } else {
            if(!islast) {
              return res
            }
        }
    }
    return res
};
```
