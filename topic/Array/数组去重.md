### 2024.5.17 数组去重 https://leetcode.cn/problems/remove-duplicates-from-sorted-array/description/

---

### JS

```js
var removeDuplicates = function(nums) {
    const setTmp = new Set()
    for(let i = 0; i<nums.length; i++) {
        if(setTmp.has(nums[i])) {
            nums.splice(i,1)
            i--
        } else {
            setTmp.add(nums[i])
        }
    }
    return nums.length
};
```
