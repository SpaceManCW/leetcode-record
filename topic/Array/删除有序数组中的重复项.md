    class Solution {
        public int removeDuplicates(int[] nums) {
            HashSet<Integer> set = new HashSet<>();
            int k = 0;
            for (int num : nums) {
                if (set.contains(num)) {
                    continue;
                }
                set.add(num);
                nums[k] = num;
                k++;
            }
            nums = set.stream().mapToInt(Integer::intValue).toArray();
            return set.size();
        }
    }
